<body>

    <div id="game-container">

        <h1>Multiplication Whac-A-Mole!</h1>

        <div id="info">

            <span id="time-left">Time: 60</span>

    <script>

        const board = document.getElementById('board');

        const scoreBoard = document.getElementById('score');

        const timeLeftBoard = document.getElementById('time-left');

        const problemBoard = document.getElementById('problem');

        const startButton = document.getElementById('start-button');


        let score = 0;

        let timeLeft = 60;

        let gameTimerId;

        let moleTimerId;

        let currentProblem = {};


        // Create the 3x3 grid for the game. [2]

        for (let i = 0; i < 9; i++) {

            let hole = document.createElement('div');

            hole.classList.add('hole');

            let mole = document.createElement('div');

            mole.classList.add('mole');

            hole.appendChild(mole);

            board.appendChild(hole);

        }


        const moles = document.querySelectorAll('.mole');


        function generateProblem() {

            const num1 = Math.floor(Math.random() * 9) + 2; // Numbers from 2 to 10

            const num2 = Math.floor(Math.random() * 9) + 2;

            const answer = num1 * num2;

            

            problemBoard.textContent = `${num1} x ${num2} = ?`;

            

            currentProblem = {

                num1,

                num2,

                answer

            };

        }


        function generateAnswers() {

            const { answer } = currentProblem;

            let answers = [answer];


            // Generate unique incorrect answers

            while (answers.length < 5) {

                const num1 = currentProblem.num1;

                const num2 = currentProblem.num2;

                let wrongAnswer;


                // Create believable wrong answers

                if (Math.random() > 0.5) {

                    wrongAnswer = (num1 + (Math.random() > 0.5 ? 1 : -1)) * num2;

                } else {

                    wrongAnswer = num1 * (num2 + (Math.random() > 0.5 ? 1 : -1));

                }


                if (!answers.includes(wrongAnswer) && wrongAnswer > 0) {

                    answers.push(wrongAnswer);

                }

            }

            return answers.sort(() => Math.random() - 0.5); // Shuffle answers

        }



        function popUp() {

            moles.forEach(mole => {

                mole.classList.remove('up');

                mole.textContent = '';

                mole.removeEventListener('click', handleMoleClick);

            });

            

            generateProblem();

            const answers = generateAnswers();

            let availableMoles = Array.from(moles);


            answers.forEach(ans => {

                if (availableMoles.length > 0) {

                    const randomIndex = Math.floor(Math.random() * availableMoles.length);

                    const mole = availableMoles.splice(randomIndex, 1)[0];

                    

                    mole.textContent = ans;

                    mole.dataset.answer = ans; // Store the answer in a data attribute

                    mole.classList.add('up');

                    mole.addEventListener('click', handleMoleClick);

                }

            });

        }


        function handleMoleClick(event) {

            const clickedAnswer = parseInt(event.target.dataset.answer, 10);

            

            if (clickedAnswer === currentProblem.answer) {

                score++;

                scoreBoard.textContent = `Score: ${score}`;

                popUp(); // Show a new problem immediately

            } else {

                score = Math.max(0, score - 1); // Decrease score for wrong answer

                scoreBoard.textContent = `Score: ${score}`;

            }

        }

        

        function countdown() {

            timeLeft--;

            timeLeftBoard.textContent = `Time: ${timeLeft}`;


            if (timeLeft === 0) {

                clearInterval(gameTimerId);

                clearInterval(moleTimerId);

                problemBoard.textContent = `Game Over! Final Score: ${score}`;

                moles.forEach(mole => mole.classList.remove('up'));

            }

        }


        function startGame() {

            score = 0;

            timeLeft = 60;

            scoreBoard.textContent = `Score: ${score}`;

            timeLeftBoard.textContent = `Time: ${timeLeft}`;

            startButton.disabled = true;


            popUp();

            moleTimerId = setInterval(popUp, 3000); // Moles pop up every 3 seconds

            gameTimerId = setInterval(countdown, 1000); // Timer counts down every second

        }


        startButton.addEventListener('click', startGame);
